258  Returns:

278 async def create_organisation(
279  db: Session,
280  org_model: OrganisationIn,
281  current_user: AuthUser,
282  logo: Optional[UploadFile] = File(None),
283 ) -> db_models.DbOrganisation:
284  """Creates a new organisation with the given name, description, url, type, and logo.
285 
286  Saves the logo file S3 bucket under /{org_id}/logo.png.
287 
288  Args:
289  db (Session): database session
290  org_model (OrganisationIn): Pydantic model for organisation input.
291  logo (UploadFile, optional): logo file of the organisation.
292  Defaults to File(...).
293  current_user: logged in user.
294 
295  Returns:
296  DbOrganisation: SQLAlchemy Organisation model.
297  """
298  if await get_organisation_by_name(db, org_name=org_model.name):
299  raise HTTPException(
300  status_code=HTTPStatus.CONFLICT,
301  detail=f"Organisation already exists with the name ({org_model.name})",
302  )
303 
304  # Required to check if exists on error
305  db_organisation = None
306 
307  try:
308  # Create new organisation without logo set
309  db_organisation = db_models.DbOrganisation(**org_model.model_dump())
310  db_organisation.user_id = current_user.id
311 
312  db.add(db_organisation)
313  db.commit()
314  # Refresh to get the assigned org id
315  db.refresh(db_organisation)
316 
317  # Update the logo field in the database with the correct path
318  if logo:
319  db_organisation.logo = await upload_logo_to_s3(db_organisation.id, logo)
320  db.commit()
321 
322  except Exception as e:
323  log.exception(e)
324  log.debug("Rolling back changes to db organisation")
325  # Rollback any changes
326  db.rollback()
327  # Delete the failed organisation entry
328  if db_organisation:
329  log.debug(f"Deleting created organisation ID {db_organisation.id}")
330  db.delete(db_organisation)
331  db.commit()
332  raise HTTPException(
333  status_code=400, detail=f"Error creating organisation: {e}"
334  ) from e
335 
336  return db_organisation

340  db: Session,
341  organisation: db_models.DbOrganisation,
342  values: OrganisationEdit,
343  logo: UploadFile(None),
344 ) -> db_models.DbOrganisation:
345  """Update an existing organisation database entry.
346 
347  Args:
348  db (Session): database session
349  organisation (DbOrganisation): Editing database model.
350  values (OrganisationEdit): Pydantic model for organisation edit.
351  logo (UploadFile, optional): logo file of the organisation.
352  Defaults to File(...).
353 
354  Returns:
355  DbOrganisation: SQLAlchemy Organisation model.
356  """
357  if not (updated_fields := values.dict(exclude_none=True)):
358  raise HTTPException(
359  status_code=HTTPStatus.UNPROCESSABLE_ENTITY,
360  detail=f"No values were provided to update organisation {organisation.id}",
361  )
362 
363  update_cmd = (
364  update(db_models.DbOrganisation)
365  .where(db_models.DbOrganisation.id == organisation.id)
366  .values(**updated_fields)
367  )
368  db.execute(update_cmd)
369 
370  if logo:
371  organisation.logo = await upload_logo_to_s3(organisation.id, logo)
372 
373  db.commit()
374  db.refresh(organisation)
375 
376  return organisation
