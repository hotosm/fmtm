228  def download_submission_for_task(task_id):
229  log.info(
230  f"Thread {threading.current_thread().name} - "
231  f"Downloading submission for Task ID {task_id}"
232  )
233  xform_name = f"{project_name}_form_{task_id}"
234  file = xform.getSubmissionMedia(odkid, xform_name)

240  def extract_files(zip_file_path):
241  log.info(
242  f"Thread {threading.current_thread().name} - "
243  f"Extracting files from {zip_file_path}"
244  )
245  with zipfile.ZipFile(zip_file_path, "r") as zip_file:
246  extract_dir = os.path.splitext(zip_file_path)[0]
247  zip_file.extractall(extract_dir)
248  return [os.path.join(extract_dir, f) for f in zip_file.namelist()]
249 
250  with concurrent.futures.ThreadPoolExecutor() as executor:
251  task_list = [x.id for x in project_tasks]
252 
253  # Download submissions using thread pool
254  futures = {
255  executor.submit(download_submission_for_task, task_id): task_id
256  for task_id in task_list
257  }
258 
259  files = []
260  for future in concurrent.futures.as_completed(futures):
261  task_id = futures[future]
262  try:
263  file_path = future.result()
264  files.append(file_path)
265  log.info(
266  f"Thread {threading.current_thread().name} -"
267  f" Task {task_id} - Download completed."
268  )
269  except Exception as e:
270  log.error(
271  f"Thread {threading.current_thread().name} -"
272  f" Error occurred while downloading submission for task "
273  f"{task_id}: {e}"
274  )
275 
276  # Extract files using thread pool
277  extracted_files = []
278  futures = {
279  executor.submit(extract_files, file_path): file_path for file_path in files
280  }
281  for future in concurrent.futures.as_completed(futures):
282  file_path = futures[future]
283  try:
284  extracted_files.extend(future.result())
285  log.info(
286  f"Thread {threading.current_thread().name} -"
287  f" Extracted files from {file_path}"
288  )
289  except Exception as e:
290  log.error(
291  f"Thread {threading.current_thread().name} -"
292  f" Error occurred while extracting files from {file_path}: {e}"
293  )
294 
295  # Create a new ZIP file for the extracted files
296  final_zip_file_path = f"{project_name}_submissions_final.zip"
297  with zipfile.ZipFile(final_zip_file_path, mode="w") as final_zip_file:
298  for file_path in extracted_files:
299  final_zip_file.write(file_path)
300 
301  return final_zip_file_path

304 def update_submission_in_s3(
305  db: Session, project_id: int, background_task_id: uuid.UUID
306 ):
307  """Update or create new submission JSON in S3 for a project."""
308  try:
309  # Get Project
310  get_project_sync = async_to_sync(project_crud.get_project)
311  project = get_project_sync(db, project_id)
312 
313  # Gather metadata
314  odk_sync = async_to_sync(project_deps.get_odk_credentials)
315  odk_credentials = odk_sync(db, project_id)
316  odk_forms = list_odk_xforms(project.odkid, odk_credentials, True)
317 
318  if not odk_forms:
319  msg = f"No odk forms returned for project ({project_id})"
320  log.warning(msg)
321  return HTTPException(
322  status_code=HTTPStatus.UNPROCESSABLE_ENTITY,
323  detail=msg,
324  )
325 
326  # Get latest submission date
327  valid_datetimes = [
328  form["lastSubmission"]
329  for form in odk_forms
330  if form["lastSubmission"] is not None
331  ]
332 
333  review_states = [
334  form["reviewStates"]
335  for form in odk_forms
336  if form["reviewStates"] is not None
337  ]
338 
339  last_submission = (
340  max(
341  valid_datetimes,
342  key=lambda x: datetime.strptime(x, "%Y-%m-%dT%H:%M:%S.%fZ"),
343  )
344  if valid_datetimes
345  else None
346  )
347  if not last_submission:
348  msg = f"Could not identify last submission for project ({project_id})"
349  log.warning(msg)
350  return HTTPException(
351  status_code=HTTPStatus.UNPROCESSABLE_ENTITY,
352  detail=msg,
353  )
354 
355  # Check if the file already exists in s3
356  s3_project_path = f"/{project.organisation_id}/{project_id}"
357  metadata_s3_path = f"/{s3_project_path}/submissions.meta.json"
358  try:
359  # Get the last submission date and review state from the metadata
360  data = get_obj_from_bucket(settings.S3_BUCKET_NAME, metadata_s3_path)
361  submission_metadata = json.loads(data.getvalue())
362 
363  zip_file_last_submission = submission_metadata["last_submission"]
364  zip_file_review_states = (
365  submission_metadata["review_states"]
366  if "review_states" in submission_metadata
367  else None
368  )
369 
370  # converting list into hash str to compare them easily
371  odk_review_state_hash = hashlib.sha256(
372  json.dumps(review_states).encode()
373  ).hexdigest()
374  s3_review_state_hash = hashlib.sha256(
375  json.dumps(zip_file_review_states).encode()
376  ).hexdigest()
377 
378  if (
379  last_submission <= zip_file_last_submission
380  and odk_review_state_hash == s3_review_state_hash
381  ):
382  # Update background task status to COMPLETED
383  update_bg_task_sync = async_to_sync(
384  project_crud.update_background_task_status_in_database
385  )
386  update_bg_task_sync(db, background_task_id, 4) # 4 is COMPLETED
387  return
388 
389  except Exception as e:
390  log.warning(str(e))
391 
392  # Zip file is outdated, regenerate
393  metadata = {
394  "last_submission": last_submission,
395  "review_states": review_states,
396  }
397 
398  # Get submissions from ODK Central
399  submissions = get_all_submissions_json(db, project_id)
400 
401  submissions_zip = BytesIO()
402  # Create a sozipfile with metadata and submissions
403  with zipfile.ZipFile(
404  submissions_zip,
405  "w",
406  compression=zipfile.ZIP_DEFLATED,
407  chunk_size=zipfile.SOZIP_DEFAULT_CHUNK_SIZE,
408  ) as myzip:
409  myzip.writestr("submissions.json", json.dumps(submissions))
410 
411  # Add zipfile to the s3 bucket
412  add_obj_to_bucket(
413  settings.S3_BUCKET_NAME,
414  submissions_zip,
415  f"/{s3_project_path}/submission.zip",
416  )
417 
418  # Upload metadata to s3
419  metadata_obj = BytesIO(json.dumps(metadata).encode())
420  add_obj_to_bucket(
421  settings.S3_BUCKET_NAME,
422  metadata_obj,
423  metadata_s3_path,
424  )
425 
426  # Update background task status to COMPLETED
427  update_bg_task_sync = async_to_sync(
428  project_crud.update_background_task_status_in_database
429  )
430  update_bg_task_sync(db, background_task_id, 4) # 4 is COMPLETED
431 
432  return True
433 
434  except Exception as e:
435  log.warning(str(e))
436  # Update background task status to FAILED
437  update_bg_task_sync = async_to_sync(
438  project_crud.update_background_task_status_in_database
439  )
440  update_bg_task_sync(db, background_task_id, 2, str(e)) # 2 is FAILED
