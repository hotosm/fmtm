61 @lru_cache(maxsize=None)
62 async def get_odk_credentials(db: Session, project_id: int):
63  """Get ODK credentials of a project, or default organization credentials."""
64  sql = text(
65  """
66  SELECT
67  COALESCE(
68  NULLIF(projects.odk_central_url, ''),
69  organisations.odk_central_url)
70  AS odk_central_url,
71  COALESCE(
72  NULLIF(projects.odk_central_user, ''),
73  organisations.odk_central_user)
74  AS odk_central_user,
75  COALESCE(
76  NULLIF(projects.odk_central_password, ''),
77  organisations.odk_central_password
78  ) AS odk_central_password
79  FROM
80  projects
81  LEFT JOIN
82  organisations ON projects.organisation_id = organisations.id
83  WHERE
84  projects.id = :project_id
85  """
86  )
87  result = db.execute(sql, {"project_id": project_id})
88  creds = result.first()
89 
90  url = creds.odk_central_url
91  user = creds.odk_central_user
92  password = creds.odk_central_password
93 
94  log.debug(f"Retrieved ODK creds for project ({project_id}): {url} | {user}")
95 
96  return project_schemas.ODKCentralDecrypted(
97  odk_central_url=url,
98  odk_central_user=user,
99  odk_central_password=password,
100  )
