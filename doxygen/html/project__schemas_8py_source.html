88  @field_validator("hashtags", mode="before")
89  @classmethod
90  def validate_hashtags(
91  cls,
92  hashtags: Optional[str | list[str]],
93  ) -> Optional[list[str]]:
94  """Validate hashtags.
95 
96  - Receives a string and parsed as a list of tags.
97  - Commas or semicolons are replaced with spaces before splitting.
98  - Add '#' to hashtag if missing.
99  """
100  if hashtags is None:
101  return None
102 
103  if isinstance(hashtags, str):
104  hashtags = hashtags.replace(",", " ").replace(";", " ")
105  hashtags_list = hashtags.split()
106  else:
107  hashtags_list = hashtags
108 
109  # Add '#' to hashtag strings if missing
110  return [
111  f"#{hashtag}" if hashtag and not hashtag.startswith("#") else hashtag
112  for hashtag in hashtags_list
113  ]
