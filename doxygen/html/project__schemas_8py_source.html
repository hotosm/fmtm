198  # Add '#' to hashtag strings if missing
199  hashtags_with_hash = [
200  f"#{hashtag}" if hashtag and not hashtag.startswith("#") else hashtag
201  for hashtag in hashtags_list
202  ]
203 
204  if "#FMTM" not in hashtags_with_hash:
205  hashtags_with_hash.append("#FMTM")

419  @field_serializer("last_active")
420  def get_last_active(self, value, values):
421  """Date of last activity on project."""
422  if value is None:
423  return None
424 
425  last_active = parser.parse(value).replace(tzinfo=None)
426  current_date = datetime.now()
427 
428  time_difference = current_date - last_active
429 
430  days_difference = time_difference.days
431 
432  if days_difference == 0:
433  return "today"
434  elif days_difference == 1:
435  return "yesterday"
436  elif days_difference < 7:
437  return f'{days_difference} day{"s" if days_difference > 1 else ""} ago'
438  else:
439  return last_active.strftime("%d %b %Y")
