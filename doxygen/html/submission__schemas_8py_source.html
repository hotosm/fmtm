80  @field_serializer("last_active")
81  def get_last_active(self, last_active: Optional[AwareDatetime]) -> str:
82  """Date of last activity on project."""
83  if last_active is None:
84  return None
85  if isinstance(last_active, str):
86  return last_active
87 
88  current_date = timestamp()
89  time_difference = current_date - last_active
90  days_difference = time_difference.days
91 
92  if days_difference == 0:
93  return "today"
94  elif days_difference == 1:
95  return "yesterday"
96  elif days_difference < 7:
97  return f'{days_difference} day{"s" if days_difference > 1 else ""} ago'
98  else:
99  return last_active.strftime("%d %b %Y")
