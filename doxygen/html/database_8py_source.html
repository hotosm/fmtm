40 async def db_conn(request: Request) -> Connection:
41  """Get a connection from the psycopg pool.
42 
43  Info on connections vs cursors:
44  https://www.psycopg.org/psycopg3/docs/advanced/async.html
45 
46  Here we are getting a connection from the pool, which will be returned
47  after the session ends / endpoint finishes processing.
48 
49  In summary:
50  - Connection is created on endpoint call.
51  - Cursors are used to execute commands throughout endpoint.
52  Note it is possible to create multiple cursors from the connection,
53  but all will be executed in the same db 'transaction'.
54  - Connection is closed on endpoint finish.
55 
56  -----------------------------------
57  To use the connection in endpoints:
58  -----------------------------------
59 
60  @app.get("/something")
61  async def do_stuff(db = Depends(db_conn)):
62  async with db.cursor() as cursor:
63  await cursor.execute("SELECT * FROM items")
64  result = await cursor.fetchall()
65  return result
66 
67  -----------------------------------
68  Additionally, the connection could be passed through to a function to
69  utilise the Pydantic model serialisation on the cursor:
70  -----------------------------------
71 
72  from psycopg.rows import class_row
73  async def get_user_by_id(db: Connection, id: int):
74  async with db.cursor(row_factory=class_row(User)) as cur:
75  await cur.execute(
76  '''
77  SELECT id, first_name, last_name, dob
78  FROM (VALUES
79  (1, 'John', 'Doe', '2000-01-01'::date),
80  (2, 'Jane', 'White', NULL)
81  ) AS data (id, first_name, last_name, dob)
82  WHERE id = %(id)s;
83  ''',
84  {"id": id},
85  )
86  obj = await cur.fetchone()
87 
88  # reveal_type(obj) would return 'Optional[User]' here
89 
90  if not obj:
91  raise KeyError(f"user {id} not found")
92 
93  # reveal_type(obj) would return 'User' here
94 
95  return obj
96  """
97  db_pool = cast(AsyncConnectionPool, request.state.db_pool)
98  async with db_pool.connection() as conn:
99  yield conn
