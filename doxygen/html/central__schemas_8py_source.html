154 def entity_fields_to_list(properties: list[str]) -> list[str]:
155  """Converts a list of Field objects to a list of field names."""
156  sanitized_properties = []
157  for property in properties:
158  if not is_valid_property_name(property):
159  log.warning(f"Invalid property name: {property},Excluding from properties.")
160  continue
161  sanitized_properties.append(sanitize_key(property))
162  default_properties = [field.name for field in ENTITY_FIELDS]
163  for item in default_properties:
164  if item not in sanitized_properties:
165  sanitized_properties.append(item)
166  return sanitized_properties
