29 async def get_project_task_activity(
30  db: Connection,
31  project_id: int,
32  days: int,
33 ) -> task_schemas.TaskEventCount:
34  """Get number of tasks mapped and validated for project.
35 
36  Args:
37  project_id (int): The ID of the project.
38  days (int): The number of days to consider for the
39  task activity (default: 10).
40  db (Connection): The database connection.
41 
42  Returns:
43  list[task_schemas.TaskEventCount]: A list of task event counts.
44  """
45  end_date = timestamp() - timedelta(days=days)
46 
47  sql = """
48  SELECT
49  to_char(created_at::date, 'dd/mm/yyyy') as date,
50  COUNT(*) FILTER (WHERE state = 'UNLOCKED_DONE') AS validated,
51  COUNT(*) FILTER (WHERE state = 'UNLOCKED_TO_VALIDATE') AS mapped
52  FROM
53  task_events
54  WHERE
55  project_id = %(project_id)s
56  AND created_at >= %(end_date)s
57  GROUP BY
58  created_at::date
59  ORDER BY
60  created_at::date;
61  """
62 
63  async with db.cursor(row_factory=class_row(task_schemas.TaskEventCount)) as cur:
64  await cur.execute(sql, {"project_id": project_id, "end_date": end_date})
65  return await cur.fetchall()
