389 async def convert_geojson_to_odk_csv(
390  input_geojson: BytesIO,
391 ) -> StringIO:
392  """Convert GeoJSON features to ODK CSV format.
393 
394  Used for form upload media (dataset) in ODK Central.
395 
396  Args:
397  input_geojson (BytesIO): GeoJSON file to convert.
398 
399  Returns:
400  feature_csv (StringIO): CSV of features in XLSForm format for ODK.
401  """
402  parsed_geojson = parse_geojson_file_to_featcol(input_geojson.getvalue())
403 
404  if not parsed_geojson:
405  raise HTTPException(
406  status_code=HTTPStatus.UNPROCESSABLE_ENTITY,
407  detail="Conversion GeoJSON --> CSV failed",
408  )
409 
410  csv_buffer = StringIO()
411  csv_writer = csv.writer(csv_buffer)
412 
413  header = ["osm_id", "tags", "version", "changeset", "timestamp", "geometry"]
414  csv_writer.writerow(header)
415 
416  features = parsed_geojson.get("features", [])
417  for feature in features:
418  geometry = feature.get("geometry")
419  javarosa_geom = await geojson_to_javarosa_geom(geometry)
420 
421  properties = feature.get("properties", {})
422  osm_id = properties.get("osm_id")
423  tags = properties.get("tags")
424  version = properties.get("version")
425  changeset = properties.get("changeset")
426  timestamp = properties.get("timestamp")
427 
428  csv_row = [osm_id, tags, version, changeset, timestamp, javarosa_geom]
429  csv_writer.writerow(csv_row)
430 
431  # Reset buffer position to start to .read() works
432  csv_buffer.seek(0)
433 
434  return csv_buffer

437 def flatten_json(data: dict, target: dict):
438  """Flatten json properties to a single level.
439 
440  Removes any existing GeoJSON data from captured GPS coordinates in
441  ODK submission.
442 
443  Usage:
444  new_dict = {}
445  flatten_json(original_dict, new_dict)
446  """
447  for k, v in data.items():
448  if isinstance(v, dict):
449  if "type" in v and "coordinates" in v:
450  # GeoJSON object found, skip it
451  continue
452  flatten_json(v, target)
453  else:
454  target[k] = v

457 async def convert_odk_submission_json_to_geojson(
458  input_json: Union[BytesIO, list],
459 ) -> geojson.FeatureCollection:
460  """Convert ODK submission JSON file to GeoJSON.
461 
462  Used for loading into QGIS.
463 
464  Args:
465  input_json (BytesIO): ODK JSON submission list.
466 
467  Returns:
468  geojson (BytesIO): GeoJSON format ODK submission.
469  """
470  if isinstance(input_json, list):
471  submission_json = input_json
472  else:
473  submission_json = json.loads(input_json.getvalue())
474 
475  if not submission_json:
476  raise HTTPException(
477  status_code=HTTPStatus.UNPROCESSABLE_ENTITY,
478  detail="Project contains no submissions yet",
479  )
480 
481  all_features = []
482  for submission in submission_json:
483  keys_to_remove = ["meta", "__id", "__system"]
484  for key in keys_to_remove:
485  submission.pop(key)
486 
487  data = {}
488  flatten_json(submission, data)
489 
490  geojson_geom = await javarosa_to_geojson_geom(
491  data.pop("xlocation", {}), geom_type="Polygon"
492  )
493 
494  feature = geojson.Feature(geometry=geojson_geom, properties=data)
495  all_features.append(feature)
496 
497  return geojson.FeatureCollection(features=all_features)

500 async def feature_geojson_to_entity_dict(
501  feature: geojson.Feature,
502 ) -> central_schemas.EntityDict:
503  """Convert a single GeoJSON to an Entity dict for upload."""
504  if not isinstance(feature, (dict, geojson.Feature)):
505  log.error(f"Feature not in correct format: {feature}")
506  raise ValueError(f"Feature not in correct format: {type(feature)}")
507 
508  feature_id = feature.get("id")
509 
510  geometry = feature.get("geometry", {})
511  if not geometry:
512  msg = "'geometry' data field is mandatory"
513  log.debug(msg)
514  raise ValueError(msg)
515 
516  javarosa_geom = await geojson_to_javarosa_geom(geometry)
517 
518  # NOTE all properties MUST be string values for Entities, convert
519  properties = {
520  str(key): str(value) for key, value in feature.get("properties", {}).items()
521  }
522  # Set to TaskStatus enum READY value (0)
523  properties["status"] = "0"
524 
525  task_id = properties.get("task_id")
526  entity_label = f"Task {task_id} Feature {feature_id}"
527 
528  return {"label": entity_label, "data": {"geometry": javarosa_geom, **properties}}
