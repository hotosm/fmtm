61  """Get all projects, or a filtered subset."""
62  bbox_condition = ""
63  bbox_params = {}
64 
65  if bbox:
66  minx, miny, maxx, maxy = map(float, bbox.split(","))
67  bbox_condition = """
68  AND ST_Intersects(
69  p.outline, ST_MakeEnvelope(%(minx)s, %(miny)s, %(maxx)s, %(maxy)s, 4326)
70  )
71  """
72  bbox_params = {"minx": minx, "miny": miny, "maxx": maxx, "maxy": maxy}
73 
74  # FIXME add logic for percentMapped and percentValidated
75  # NOTE alternative logic to build a FeatureCollection
76  """
77  SELECT jsonb_build_object(
78  'type', 'FeatureCollection',
79  'features', COALESCE(jsonb_agg(feature), '[]'::jsonb)
80  ) AS featcol
81  FROM (...
82  """
83  sql = f"""
84  SELECT
85  'FeatureCollection' as type,
86  COALESCE(jsonb_agg(feature), '[]'::jsonb) AS features
87  FROM (
88  SELECT jsonb_build_object(
89  'type', 'Feature',
90  'id', p.id,
91  'geometry', ST_AsGeoJSON(p.outline)::jsonb,
92  'properties', jsonb_build_object(
93  'name', p.name,

95  'percentValidated', 0,
96  'created', p.created_at,
97  'link', concat('https://', %(domain)s::text, '/project/', p.id)
98  )
99  ) AS feature
100  FROM projects p
101  WHERE p.visibility = 'PUBLIC'
102  {bbox_condition}
103  ) AS features;
104  """

189 async def read_and_insert_xlsforms(db: Connection, directory: str) -> None:
190  """Read the list of XLSForms from the disk and sync them with the database."""
191  async with db.cursor() as cur:
192  existing_db_forms = set()
193 
194  # Collect all existing XLSForm titles from the database
195  select_existing_query = """
196  SELECT title FROM xlsforms;
197  """
198  await cur.execute(select_existing_query)
199  existing_db_forms = {row[0] for row in await cur.fetchall()}
200 
201  # Insert or update new XLSForms from disk
202  for xls_type in XLSFormType:
203  file_name = xls_type.name
204  category = xls_type.value
205  file_path = Path(directory) / f"{file_name}.xls"
206 
207  if not file_path.exists():
208  log.warning(f"{file_path} does not exist!")
209  continue
210 
211  if file_path.stat().st_size == 0:
212  log.warning(f"{file_path} is empty!")
213  continue
214 
215  with open(file_path, "rb") as xls:
216  data = xls.read()
217 
218  try:
219  insert_query = """
220  INSERT INTO xlsforms (title, xls)
221  VALUES (%(title)s, %(xls)s)
222  ON CONFLICT (title) DO UPDATE
223  SET xls = EXCLUDED.xls
224  """
225  await cur.execute(insert_query, {"title": category, "xls": data})
226  log.info(f"XLSForm for '{category}' inserted/updated in the database")
227 
228  except Exception as e:
229  log.exception(
230  f"Failed to insert or update {category} in the database. "
231  f"Error: {e}",
232  stack_info=True,
233  )
234 
235  # Determine the forms that need to be deleted (those in the DB but
236  # not in the current XLSFormType)
237  required_forms = {xls_type.value for xls_type in XLSFormType}
238  forms_to_delete = existing_db_forms - required_forms
239 
240  if forms_to_delete:
241  delete_query = """
242  DELETE FROM xlsforms WHERE title = ANY(%(titles)s)
243  """
244  await cur.execute(delete_query, {"titles": list(forms_to_delete)})
245  log.info(f"Deleted XLSForms from the database: {forms_to_delete}")

386  """Determine predominant geometry type for extract."""

452 def flatten_dict(d, parent_key="", sep="_"):
453  """Recursively flattens a nested dictionary into a single-level dictionary.
454 
455  Args:
456  d (dict): The input dictionary.
457  parent_key (str): The parent key (used for recursion).
458  sep (str): The separator character to use in flattened keys.
459 
460  Returns:
461  dict: The flattened dictionary.
462  """
463  items = {}
464  for k, v in d.items():
465  new_key = f"{parent_key}{sep}{k}" if parent_key else k
466  if isinstance(v, dict):
467  items.update(flatten_dict(v, new_key, sep=sep))
468  else:
469  items[new_key] = v
470  return items
