96  .limit(limit)
97  .all()
98  )
99  project_count = db.query(db_models.DbProject).filter(and_(*filters)).count()
100 
101  else:
102  db_projects = (
103  db.query(db_models.DbProject)
104  .filter(
105  db_models.DbProject.visibility # type: ignore
106  == ProjectVisibility.PUBLIC # type: ignore
107  )
108  .order_by(db_models.DbProject.id.desc()) # type: ignore
109  .offset(skip)
110  .limit(limit)
111  .all()
112  )
113  project_count = db.query(db_models.DbProject).count()
114 
115  # TODO refactor to use Pydantic model methods instead.
116  if db_projects and len(db_projects) > 0:
117 
118  async def convert_project(project):
119  return await convert_to_app_project(project)

131 async def get_projects_featcol(
132  db: Session,
133  bbox: Optional[str] = None,
134 ) -> geojson.FeatureCollection:
135  """Get all projects, or a filtered subset."""
136  bbox_condition = (
137  """AND ST_Intersects(
138  p.outline, ST_MakeEnvelope(:minx, :miny, :maxx, :maxy, 4326)
139  )"""
140  if bbox
141  else ""
142  )
143 
144  bbox_params = {}
145  if bbox:
146  minx, miny, maxx, maxy = map(float, bbox.split(","))
147  bbox_params = {"minx": minx, "miny": miny, "maxx": maxx, "maxy": maxy}
148 
149  query = text(
150  f"""
151  SELECT jsonb_build_object(
152  'type', 'FeatureCollection',
153  'features', COALESCE(jsonb_agg(feature), '[]'::jsonb)
154  ) AS featcol
155  FROM (
156  SELECT jsonb_build_object(
157  'type', 'Feature',
158  'id', p.id,
159  'geometry', ST_AsGeoJSON(p.outline)::jsonb,
160  'properties', jsonb_build_object(
161  'name', pi.name,
162  'percentMapped', 0,
163  'percentValidated', 0,
164  'created', p.created_at,
165  'link', concat('https://', :domain, '/project/', p.id)
166  )
167  ) AS feature
168  FROM projects p
169  LEFT JOIN project_info pi ON p.id = pi.project_id
170  WHERE p.visibility = 'PUBLIC'
171  {bbox_condition}
172  ) features;
173  """
174  )
175 
176  result = db.execute(query, {"domain": settings.FMTM_DOMAIN, **bbox_params})
177  featcol = result.scalar_one()
178 
179  return featcol

393 ):
394  """Create tasks for a project, from provided task boundaries."""
395  try:
396  if isinstance(boundaries, str):
397  boundaries = json.loads(boundaries)
398 
399  # Update the boundary polyon on the database.
400  if boundaries["type"] == "Feature":
401  polygons = [boundaries]
402  else:
403  polygons = boundaries["features"]
404  log.debug(f"Processing {len(polygons)} task geometries")
405  for index, polygon in enumerate(polygons):
406  # If the polygon is a MultiPolygon, convert it to a Polygon
407  if polygon["geometry"]["type"] == "MultiPolygon":
408  log.debug("Converting MultiPolygon to Polygon")
409  polygon["geometry"]["type"] = "Polygon"
410  polygon["geometry"]["coordinates"] = polygon["geometry"]["coordinates"][
411  0
412  ]
413 
414  db_task = db_models.DbTask(
415  project_id=project_id,
416  outline=wkblib.dumps(shape(polygon["geometry"]), hex=True),
417  project_task_index=index + 1,
418  )
419  db.add(db_task)
420  log.debug(
421  "Created database task | "
422  f"Project ID {project_id} | "
423  f"Task index {index}"
424  )
425 
426  # Commit all tasks and update project location in db
427  db.commit()
428 
429  log.debug("COMPLETE: creating project boundary, based on task boundaries")
430 
431  return True
432  except Exception as e:
433  log.exception(e)
434  raise HTTPException(HTTPStatus.UNPROCESSABLE_ENTITY, detail=e) from e

560 async def read_and_insert_xlsforms(db, directory) -> None:
561  """Read the list of XLSForms from the disk and insert to DB."""
562  # Insert new XLSForms to DB and update existing ones
563  for xls_type in XLSFormType:
564  file_name = xls_type.name
565  category = xls_type.value
566  file_path = Path(directory) / f"{file_name}.xls"
567 
568  if not file_path.exists():
569  log.warning(f"{file_path} does not exist!")
570  continue
571 
572  if file_path.stat().st_size == 0:
573  log.warning(f"{file_path} is empty!")
574  continue
575 
576  with open(file_path, "rb") as xls:
577  data = xls.read()
578 
579  try:
580  insert_query = text(
581  """
582  INSERT INTO xlsforms (title, xls)
583  VALUES (:title, :xls)
584  ON CONFLICT (title) DO UPDATE SET
585  title = EXCLUDED.title, xls = EXCLUDED.xls
586  """
587  )
588  db.execute(insert_query, {"title": category, "xls": data})
589  db.commit()
590  log.info(f"XLSForm for '{category}' present in the database")
591 
592  except Exception as e:
593  log.error(
594  f"Failed to insert or update {category} in the database. Error: {e}"
595  )
596 
597  existing_db_forms = {
598  title for (title,) in db.query(db_models.DbXLSForm.title).all()
599  }
600  required_forms = {xls_type.value for xls_type in XLSFormType}
601  # Delete XLSForms from DB that are not found on disk
602  for title in existing_db_forms - required_forms:
603  delete_query = text(
604  """
605  DELETE FROM xlsforms WHERE title = :title
606  """
607  )
608  db.execute(delete_query, {"title": title})
609  db.commit()
610  log.info(
611  f"Deleted {title} from the database as it was not present in XLSFormType."
612  )

842 def flatten_dict(d, parent_key="", sep="_"):
843  """Recursively flattens a nested dictionary into a single-level dictionary.
844 
845  Args:
846  d (dict): The input dictionary.
847  parent_key (str): The parent key (used for recursion).
848  sep (str): The separator character to use in flattened keys.
849 
850  Returns:
851  dict: The flattened dictionary.
852  """
853  items = {}
854  for k, v in d.items():
855  new_key = f"{parent_key}{sep}{k}" if parent_key else k
856  if isinstance(v, dict):
857  items.update(flatten_dict(v, new_key, sep=sep))
858  else:
859  items[new_key] = v
860  return items
