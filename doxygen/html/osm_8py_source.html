51 async def handle_osm_callback(request: Request, osm_auth: Auth):
52  """Handle OSM callback in OAuth flow.
53 
54  Args:
55  request: The GET request.
56  osm_auth: The Auth object from osm-login-python.
57 
58  Returns:
59  Response: A response including cookies that will be set in-browser.
60  """
61  log.debug(f"Callback url requested: {request.url}")
62 
63  # Enforce https callback url for openstreetmap.org
64  callback_url = str(request.url).replace("http://", "https://")
65 
66  # Get user data from response
67  try:
68  tokens = osm_auth.callback(callback_url)
69  serialised_user_data = tokens.get("user_data")
70  log.debug(f"Access token returned of length {len(serialised_user_data)}")
71  osm_user = osm_auth.deserialize_data(serialised_user_data)
72  user_data = {
73  "sub": f"fmtm|{osm_user['id']}",
74  "aud": settings.FMTM_DOMAIN,
75  "iat": int(time()),
76  "exp": int(time()) + 86400, # expiry set to 1 day
77  "username": osm_user["username"],
78  "email": osm_user.get("email"),
79  "picture": osm_user.get("img_url"),
80  "role": UserRole.MAPPER,
81  }
82  except Exception as e:
83  raise ValueError(f"Invalid OSM token: {e}") from e
84 
85  # Create our JWT tokens from user data
86  fmtm_token, refresh_token = create_jwt_tokens(user_data)
87  response = Response(status_code=HTTPStatus.OK)
88  response_plus_cookies = set_cookies(response, fmtm_token, refresh_token)
89 
90  # NOTE Here we create a separate cookie to store the OSM token, for later
91  # workflows such as conflation (OSM changesets) or OSM messaging.
92  # First, we get the OSM token from response.
93  # The token is serialised in the cookie & must be deserialised using
94  # osm_login_python.auth.deserialize_data.
95  serialised_osm_token = tokens.get("oauth_token")
96  cookie_name = settings.cookie_name
97  osm_cookie_name = f"{cookie_name}_osm"
98  log.debug(f"Creating cookie '{osm_cookie_name}' with OSM token")
99  response_plus_cookies.set_cookie(
100  key=osm_cookie_name,
101  value=serialised_osm_token,
102  max_age=864000,
103  expires=864000, # expiry set for 10 days
104  path="/",
105  domain=settings.FMTM_DOMAIN,
106  secure=False if settings.DEBUG else True,
107  httponly=True,
108  samesite="lax",
109  )
110 
111  return response_plus_cookies

114 def get_osm_token(request: Request, osm_auth: Auth) -> str:
115  """Extract and deserialize OSM token from cookies."""
116  cookie_name = f"{settings.cookie_name}_osm"
117  log.debug(f"Extracting OSM token from cookie {cookie_name}")
118  serialised_osm_token = request.cookies.get(cookie_name)
119  if not serialised_osm_token:
120  raise HTTPException(
121  status_code=HTTPStatus.UNAUTHORIZED,
122  detail="You must be logged in to your OpenStreetMap account.",
123  )
124  return osm_auth.deserialize_data(serialised_osm_token)

127 def send_osm_message(
128  osm_token: str,
129  title: str,
130  body: str,
131  osm_username: str = None,
132  osm_id: int = None,
133 ) -> None:
134  """Send a message via OSM API."""
135  if not osm_username and not osm_id:
136  raise ValueError("Either recipient or recipient_id must be provided.")
137 
138  email_url = f"{settings.OSM_URL}api/0.6/user/messages"
139  headers = {"Authorization": f"Bearer {osm_token}"}
140  post_body = {
141  "title": title,
142  "body": body,
143  }
144 
145  if osm_id:
146  post_body["recipient_id"] = osm_id
147  else:
148  post_body["recipient"] = osm_username
149 
150  log.debug(
151  f"Sending message to user ({osm_id or osm_username}) via OSM API: {email_url}"
152  )
153  response = requests.post(email_url, headers=headers, data=post_body)
154 
155  if response.status_code == 200:
156  log.info("Message sent successfully")
157  else:
158  msg = "Sending message via OSM failed"
159  log.error(f"{msg}: {response.text}")
