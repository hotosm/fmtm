164  @classmethod
165  async def one(cls, db: Connection, user_identifier: int | str) -> Self:
166  """Get a user either by ID or username, including roles and orgs managed."""
167  async with db.cursor(row_factory=class_row(cls)) as cur:
168  sql = """
169  SELECT
170  u.*,
171 
172  -- Aggregate organisation IDs managed by the user
173  array_agg(
174  DISTINCT om.organisation_id
175  ) FILTER (WHERE om.organisation_id IS NOT NULL) AS orgs_managed,
176 
177  -- Aggregate project roles for the user, as project:role pairs
178  jsonb_object_agg(
179  ur.project_id,
180  COALESCE(ur.role, 'MAPPER')
181  ) FILTER (WHERE ur.project_id IS NOT NULL) AS project_roles
182 
183  FROM users u
184  LEFT JOIN user_roles ur ON u.id = ur.user_id
185  LEFT JOIN organisation_managers om ON u.id = om.user_id
186  """
187 
188  if isinstance(user_identifier, int):
189  # Is ID
190  sql += """
191  WHERE u.id = %(user_identifier)s
192  GROUP BY u.id;
193  """
194  else:
195  # Is username (basic flexible matching)
196  sql += """
197  WHERE u.username ILIKE %(user_identifier)s
198  GROUP BY u.id;
199  """
200  user_identifier = f"{user_identifier}%"
201 
202  await cur.execute(
203  sql,
204  {"user_identifier": user_identifier},
205  )
206  db_project = await cur.fetchone()
207

670  project_id: Optional[int] = None,
671  task_id: Optional[int] = None,
672  days: Optional[int] = None,
673  comments: Optional[bool] = None,
674  ) -> Optional[list[Self]]:
675  """Fetch all task event entries for a project.
676 
677  Args:
678  db (Connection): the database connection.
679  project_id (int): return all events for a project.
680  task_id (Connection): return all events for a task.
681  days (int): filter to only include events since X days ago.
682  comments (bool): show comments rather than events.

687  if project_id and task_id:
688  raise ValueError("Specify either project_id or task_id, not both.")

690  raise ValueError("Either project_id or task_id must be provided.")

700  params["task_id"] = task_id
701  if days is not None:
702  end_date = timestamp() - timedelta(days=days)
703  filters.append("created_at >= %(end_date)s")
704  params["end_date"] = end_date
705  if comments:
706  filters.append("event = 'COMMENT'")
707  else:
708  filters.append("event != 'COMMENT'")
709 
710  filters_joined = " AND ".join(filters)
711 
712  sql = f"""
713  SELECT
714  the.*,
715  u.profile_img
716  FROM
717  public.task_events the
718  LEFT JOIN
719  users u ON u.id = the.user_id
720  WHERE {filters_joined}
721  ORDER BY created_at DESC;
722  """
723

725  await cur.execute(sql, params)
