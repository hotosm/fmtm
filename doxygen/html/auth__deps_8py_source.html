113 def create_jwt_tokens(input_data: dict) -> tuple[str, str]:
114  """Generate access and refresh tokens.
115 
116  Args:
117  input_data (dict): user data for which the access token is being generated.
118 
119  Returns:
120  tuple[str]: The generated access tokens.
121  """
122  access_token_data = input_data.copy()
123  # Set refresh token expiry to 7 days
124  refresh_token_data = {**input_data, "exp": int(time()) + 86400 * 7}
125 
126  encryption_key = settings.ENCRYPTION_KEY.get_secret_value()
127  algorithm = settings.JWT_ENCRYPTION_ALGORITHM
128 
129  access_token = jwt.encode(access_token_data, encryption_key, algorithm=algorithm)
130  refresh_token = jwt.encode(refresh_token_data, encryption_key, algorithm=algorithm)
131 
132  return access_token, refresh_token

149 def verify_jwt_token(token: str, ignore_expiry: bool = False) -> dict:
150  """Verify the access token and return its payload.
151 
152  Args:
153  token (str): The access token to be verified.
154  ignore_expiry (bool): Do not throw an error if the token is expired
155  upon deserialisation.
156 
157  Returns:
158  dict: The payload of the access token if verification is successful.
159 
160  Raises:
161  HTTPException: If the token has expired or credentials could not be validated.
162  """
163  try:
164  return jwt.decode(
165  token,
166  settings.ENCRYPTION_KEY.get_secret_value(),
167  algorithms=[settings.JWT_ENCRYPTION_ALGORITHM],
168  audience=settings.FMTM_DOMAIN,
169  options={"verify_exp": False if ignore_expiry else True},
170  )
171  except jwt.ExpiredSignatureError as e:
172  raise HTTPException(
173  status_code=HTTPStatus.UNAUTHORIZED,
174  detail="Refresh token has expired",
175  ) from e
176  except jwt.PyJWTError as e:
177  raise HTTPException(
178  status_code=HTTPStatus.UNAUTHORIZED,
179  detail="Could not validate refresh token",
180  ) from e
181  except Exception as e:
182  log.exception(f"Unknown cookie/jwt error: {e}")
183  raise HTTPException(
184  status_code=HTTPStatus.UNAUTHORIZED,
185  detail="Could not validate refresh token",
186  ) from e

189 async def refresh_cookies(
190  request: Request,
191  current_user: AuthUser,
192  cookie_name: str,
193  refresh_cookie_name: str,
194 ):
195  """Reusable function to renew the expiry on cookies.
196 
197  Used by both management and mapper refresh endpoints.
198  """
199  if settings.DEBUG:
200  return JSONResponse(
201  status_code=HTTPStatus.OK,
202  content={**current_user.model_dump()},
203  )
204 
205  access_token = get_cookie_value(request, cookie_name)
206  if not access_token:
207  raise HTTPException(
208  status_code=HTTPStatus.UNAUTHORIZED,
209  detail="No access token provided",
210  )
211 
212  refresh_token = get_cookie_value(request, refresh_cookie_name)
213  if not refresh_token:
214  raise HTTPException(
215  status_code=HTTPStatus.UNAUTHORIZED,
216  detail="No refresh token provided",
217  )
218 
219  # Decode JWT and get data from both cookies,
220  # checking refresh expiry is valid first
221  refresh_token_data = verify_jwt_token(refresh_token)
222  access_token_data = verify_jwt_token(access_token, ignore_expiry=True)
223 
224  try:
225  # Refresh token + refresh token
226  new_access_token = refresh_jwt_token(access_token_data)
227  new_refresh_token = refresh_jwt_token(refresh_token_data)
228  except Exception as e:
229  raise HTTPException(
230  status_code=HTTPStatus.BAD_REQUEST,
231  detail=f"Failed to refresh tokens: {e}",
232  ) from e
233 
234  # NOTE Append the user data to the JSONResponse so we can display in the
235  # frontend header. For the mapper frontend this is enough, but for the
236  # management frontend we instead use the return from /auth/me
237  response = JSONResponse(status_code=HTTPStatus.OK, content=access_token_data)
238  response_plus_cookies = set_cookies(response, new_access_token, new_refresh_token)
239 
240  # Invalidate any temp cookies from mapper frontend
241  for cookie_name in [
242  f"{settings.cookie_name}_temp",
243  f"{settings.cookie_name}_temp_refresh",
244  ]:
245  log.debug(f"Resetting cookie in response named '{cookie_name}'")
246  response_plus_cookies.set_cookie(
247  key=cookie_name,
248  value="",
249  max_age=0, # Set to expire immediately
250  expires=0, # Set to expire immediately
251  path="/",
252  domain=settings.FMTM_DOMAIN,
253  secure=False if settings.DEBUG else True,
254  httponly=True,
255  samesite="lax",
256  )
257 
258  return response_plus_cookies

309 async def _authenticate_user(access_token: Optional[str]) -> AuthUser:
310  """Authenticate user by verifying the access token."""
311  if not access_token:
312  raise HTTPException(
313  status_code=HTTPStatus.UNAUTHORIZED,
314  detail="No access token provided",
315  )
316 
317  try:
318  token_data = verify_jwt_token(access_token)
319  except ValueError as e:
320  log.exception(f"Failed to verify access token: {e}", stack_info=True)
321  raise HTTPException(
322  status_code=HTTPStatus.UNAUTHORIZED,
323  detail="Access token not valid",
324  ) from e
